# createClient()

```ts
function createClient(config): Base44Client;
```

Create a Base44 client instance for interacting with the Base44 API.

This is the primary method for initializing the SDK. It creates a client with modules for entities, integrations, authentication, functions, and agents.

## Parameters

<ParamField path="config" type="object" required>
  Client configuration options

  <Expandable title="properties">
    <ParamField path="appId" type="string" required>
      Your Base44 application ID
    </ParamField>

    <ParamField path="serverUrl" type="string" default="https://base44.app">
      Base44 API server URL
    </ParamField>

    <ParamField path="appBaseUrl" type="string">
      Your application's base URL for authentication redirects
    </ParamField>

    <ParamField path="token" type="string">
      User authentication token for making authenticated requests
    </ParamField>

    <ParamField path="serviceToken" type="string">
      Service role token for elevated privileges (server-side only)
    </ParamField>

    <ParamField path="requiresAuth" type="boolean" default={false}>
      If true, automatically redirects to login when not authenticated
    </ParamField>

    <ParamField path="functionsVersion" type="string">
      Specific version of functions to use
    </ParamField>

    <ParamField path="headers" type="Record<string, string>">
      Additional headers to include in all API requests
    </ParamField>

    <ParamField path="options" type="CreateClientOptions">
      Additional client options like error handlers
    </ParamField>

  </Expandable>
</ParamField>

## Returns

<ResponseField name="Base44Client" type="object">
  A Base44 client instance with the following modules and methods

  <Expandable title="properties">
    <ResponseField name="entities" type="EntitiesModule">
      CRUD operations for entities - `list()`, `get()`, `create()`, `update()`, `delete()`, `filter()`, `bulkCreate()`, etc.
      
      See [createEntitiesModule](../../modules/entities/functions/createEntitiesModule)
    </ResponseField>

    <ResponseField name="integrations" type="IntegrationsModule">
      Call integration endpoints - Dynamic access to any integration: `client.integrations.PackageName.EndpointName()`

      See [createIntegrationsModule](../../modules/integrations/functions/createIntegrationsModule)
    </ResponseField>

    <ResponseField name="auth" type="AuthModule">
      Authentication and user management - `me()`, `updateMe()`, `loginViaEmailPassword()`, `register()`, `logout()`, etc.

      See [createAuthModule](../../modules/auth/functions/createAuthModule)
    </ResponseField>

    <ResponseField name="functions" type="FunctionsModule">
      Invoke custom cloud functions - `invoke(functionName, params)`

      See [createFunctionsModule](../../modules/functions/functions/createFunctionsModule)
    </ResponseField>

    <ResponseField name="agents" type="AgentsModule">
      AI agent operations - `getConversations()`, `createConversation()`, `addMessage()`, `subscribeToConversation()`, etc.

      See [createAgentsModule](../../modules/agents/functions/createAgentsModule)
    </ResponseField>

    <ResponseField name="appLogs" type="AppLogsModule">
      Application logging - `logUserInApp()`, `fetchLogs()`, `getStats()`
    </ResponseField>

    <ResponseField name="setToken" type="(token: string) => void">
      Update the authentication token
    </ResponseField>

    <ResponseField name="getConfig" type="() => CreateClientConfig">
      Get current client configuration
    </ResponseField>

    <ResponseField name="cleanup" type="() => void">
      Clean up resources (websockets, etc.)
    </ResponseField>

    <ResponseField name="asServiceRole" type="object">
      Access modules with elevated privileges (requires `serviceToken`). All the same modules as above but with service role permissions. Does NOT include the `auth` module for security.
    </ResponseField>

  </Expandable>
</ResponseField>

## Examples

<CodeGroup>

```typescript Basic Setup
import { createClient } from "@base44/sdk";

const client = createClient({
  appId: "your-app-id",
});
```

```typescript With Authentication
const client = createClient({
  appId: "your-app-id",
  token: "user-auth-token",
});

const user = await client.auth.me();
```

```typescript With Service Role
const client = createClient({
  appId: "your-app-id",
  token: "user-token",
  serviceToken: "service-token",
});

// User-level access
const myData = await client.entities.Product.list();

// Admin-level access
const allData = await client.asServiceRole.entities.Product.list();
```

```typescript Server-Side
const client = createClient({
  appId: process.env.BASE44_APP_ID!,
  serviceToken: process.env.BASE44_SERVICE_TOKEN!,
});
```

</CodeGroup>
